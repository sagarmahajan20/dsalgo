// Function to find subarray with maximum sum
// arr: input array
// n: size of array
int maxSubarraySum(int arr[], int n){
    
    
    int sum=0;
    int max = arr[0];
    
    for(int i=0; i<n; i++)
    {
        sum = sum +  arr[i];
       
        
        if(sum> max )
        {
            max = sum;
        }
        
         if(sum <0)
        {
            sum=0;
        }
        
        
        
    }
    
   return max;
}
